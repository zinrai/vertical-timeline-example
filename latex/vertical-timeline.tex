\documentclass[a4paper,11pt]{article}

% TikZパッケージの読み込み
\usepackage{tikz}

% タイムラインエントリを定義するマクロ
\newcommand{\timelineentry}[3]{
  \node[year, anchor=east] at (-0.2, -#1) {#2};
  \node[event, anchor=west] at (0.2, -#1) {#3};
  \draw[thick] (-0.1,-#1) -- (0.1,-#1);
}

% タイムライン全体を描画するマクロ
\newcommand{\drawtimeline}[1]{
  \begin{tikzpicture}[
    event/.style={font=\normalsize, align=left},
    year/.style={font=\bfseries}
  ]

  % タイムラインの垂直線
  \draw[thick] (0,0) -- (0,-#1);

  % タイムラインエントリ
  \timelineentry{0}{2020/02}{新型コロナウイルスのパンデミックが\\ 世界中に広がる}
  \timelineentry{3}{2021/08}{東京オリンピック・パラリンピックが\\ 1年遅れで開催される}
  \timelineentry{6}{2022/02}{ほげほげ}
  \timelineentry{9}{2022/12}{ふがふが}
  \timelineentry{12}{2023/06}{ほげほげ}

  \end{tikzpicture}
}
